2.1. T-test¶

Null Hypothesis: User-ratings and Popularity are related¶

Alternate Hypothesis: User-ratings and Popularity are not related¶

from scipy.stats import *
popular = bookMaster.Popularity
ratings = bookMaster.USERRATINGS

popular_std = np.std(popular)
ratings_std = np.std(ratings)
print("Popularity Standard Deviation: ", popular_std)
print("Ratings Standard Deviation: ", ratings_std)

ttest,pval = ttest_ind(popular, ratings)

if(pval <0.05):
    print("\n\nNull Hypothesis is rejected")
else:

b=stats.chi2_contingency(cross_tab)
Expected_Values = b[3]

no_of_columns=len(cross_tab.iloc[0,:])
ddof=(no_of_rows-1)*(no_of_columns-1)

from scipy.stats import chi2
chi_square=sum([(o-e)**2./e for o,e in zip(Observed_Values,Expected_Values)])
chi_square_statistic=chi_square[0]+chi_square[1]
print("chi-square statistic:-", chi_square_statistic)

alpha = 0.05
critical_value=chi2.ppf(q=1-alpha, df=ddof)

#p-value
